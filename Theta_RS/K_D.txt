% K_D: slowly inactivating potassium current.
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

% Parameters.
gD = 0.39
EK = -90

theta_a = -50
sigma_a = 20
tau_a = 2

theta_b = -70
sigma_b = -6
tau_b = 150

% ODEs.
a' = (inf(V,theta_a,sigma_a)-a)/tau_a
b' = (inf(V,theta_b,sigma_b)-b)/tau_b

% Functions.
inf(V,theta,sigma) = 1/(1 + exp((theta-V)/sigma))
i_D = -gD*a^3*b*(EK-V)

% Interface.
current => i_D
